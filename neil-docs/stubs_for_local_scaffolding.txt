# src/gdscript/utils/deterministic_id.gd
# GDScript: deterministic name-based UUIDv5-like using SHA256 hex (not RFC4122 binary)
extends Node

class_name DeterministicID

static func canonicalize(name: String) -> String:
    var s = name.strip_edges().to_lower()
    s = s.replace("\n", " ").replace("\r", " ")
    s = RegEx.new().sub(s, "\\s+", " ") if Engine.has_singleton("RegEx")
    return s

static func id_for(namespace: String, name: String) -> String:
    var n = canonicalize(namespace)
    var m = canonicalize(name)
    var input = "%s:%s".format(n, m)
    var hash = Crypto.digest(Crypto.HASH_SHA256, input.to_utf8())
    # Return hex prefixed with ds- for Dungeon Seed
    return "ds-" + hash.hex_encode()

# src/gdscript/utils/rng_wrapper.gd
# Simple RNG wrapper providing deterministic PRNG (Xorshift64*)
extends Node

class_name RNGWrapper

var state: int = 88172645463325252 # default seed

func seed(s: int) -> void:
    state = int(s) & 0xffffffffffffffff
    if state == 0:
        state = 1

func next_u64() -> int:
    var x = state
    x ^= (x << 13) & 0xffffffffffffffff
    x ^= (x >> 7) & 0xffffffffffffffff
    x ^= (x << 17) & 0xffffffffffffffff
    state = x
    return x & 0xffffffffffffffff

func randf() -> float:
    return float(next_u64() & 0xffffffff) / float(0xffffffff)

# tools/rng/generate_vectors.py (simple generator for fixtures)
import argparse
import json

class Xorshift64:
    def __init__(self, seed):
        self.x = seed & 0xffffffffffffffff
        if self.x == 0:
            self.x = 1
    def next_u64(self):
        x = self.x
        x ^= (x << 13) & 0xffffffffffffffff
        x ^= (x >> 7) & 0xffffffffffffffff
        x ^= (x << 17) & 0xffffffffffffffff
        self.x = x & 0xffffffffffffffff
        return self.x

if __name__ == '__main__':
    p = argparse.ArgumentParser()
    p.add_argument('--seed', type=int, default=12345)
    p.add_argument('--count', type=int, default=10)
    p.add_argument('--out', type=str, default='tests/fixtures/rng/seed-0001.json')
    args = p.parse_args()
    rng = Xorshift64(args.seed)
    vecs = [rng.next_u64() for _ in range(args.count)]
    with open(args.out, 'w') as f:
        json.dump({'seed': args.seed, 'vectors': vecs}, f, indent=2)
